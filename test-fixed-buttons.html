Тест исправленных кнопок "Добавить лицензию" и "Добавить шаблон"

Что было исправлено

  • Удалена сложная логика отслеживания состояния кнопки шаблона
  • Убрано прямое манипулирование DOM из функций addCard и addTemplate
  • Удален явный обработчик событий в onMounted, который конфликтовал с Vue
  • Сохранены диагностические функции для отладки
  • Упрощена структура компонента для соответствия принципам Vue

Инструкции по тестированию

  1. Откройте приложение в iframe ниже
  2. Откройте консоль разработчика (F12) в приложении
  3. Проверьте начальное состояние кнопок
  4. Нажмите кнопку "Добавить лицензию" (□)
  5. Проверьте, что карточка добавилась корректно
  6. Нажмите кнопку "Добавить шаблон" (⧉)
  7. Проверьте, что шаблон загрузился корректно
  8. Отметьте выполненные шаги в чек-листе ниже

Код для проверки в консоли

Скопируйте и выполните этот код в консоли приложения для дополнительной проверки:

// Проверка состояния кнопок function checkButtons() { const licenseBtn = document.querySelector('[title="Добавить лицензию"]'); const templateBtn = document.querySelector('[title="Добавить шаблон"]'); console.log('=== Проверка состояния кнопок ==='); if (licenseBtn) { console.log('Кнопка "Добавить лицензия":', { exists: true, disabled: licenseBtn.disabled, classList: Array.from(licenseBtn.classList) }); } else { console.log('Кнопка "Добавить лицензия" не найдена'); } if (templateBtn) { console.log('Кнопка "Добавить шаблон":', { exists: true, disabled: templateBtn.disabled, classList: Array.from(templateBtn.classList) }); } else { console.log('Кнопка "Добавить шаблон" не найдена'); } } // Проверка состояния хранилища function checkStoreState() { try { const app = document.querySelector('#app').__vue_app__; if (app && app.config.globalProperties.$pinia) { const pinia = app.config.globalProperties.$pinia; const cardsStore = pinia._s.get('cards'); const connectionsStore = pinia._s.get('connections'); if (cardsStore) { console.log('Состояние хранилища карточек:', { cardsCount: cardsStore.cards.length, selectedCardIds: cardsStore.selectedCardIds }); } if (connectionsStore) { console.log('Состояние хранилища соединений:', { connectionsCount: connectionsStore.connections.length }); } } } catch (e) { console.log('Не удалось получить доступ к хранилищу:', e); } } // Запуск проверок checkButtons(); checkStoreState();
